        // ===== STARS BACKGROUND =====
        const canvas = document.getElementById('stars');
        const ctx = canvas.getContext('2d');
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;

        const stars = [];
        for (let i = 0; i < 150; i++) {
            stars.push({
                x: Math.random() * canvas.width,
                y: Math.random() * canvas.height,
                radius: Math.random() * 1.5 + 0.5,
                alpha: Math.random(),
                delta: Math.random() * 0.02 + 0.01
            });
        }

        function drawStars() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            for (let star of stars) {
                ctx.beginPath();
                ctx.arc(star.x, star.y, star.radius, 0, 2*Math.PI);
                ctx.fillStyle = `rgba(255, 255, 255, ${star.alpha})`;
                ctx.fill();
                star.alpha += star.delta;
                if (star.alpha <= 0 || star.alpha >= 1) star.delta = -star.delta;
            }
            requestAnimationFrame(drawStars);
        }
        drawStars();

        // ===== CORE LOGIC =====
        const wishEl = document.getElementById("wish");
        const music = document.getElementById("music");

        const message = `Noor <span class="key-word">üíñ</span>, from the first moment I saw you, my world changed. 
Your <span class="key-word">smile</span>, your <span class="key-word">laugh</span>, your <span class="key-word">kindness</span> ‚Äî it fills every corner of my heart.

<span class="key-word">forever</span> and <span class="key-word">always</span>. üåüüíñ`;

        let index = 0;

        function typeSlowly() {
            if (index < message.length) {
                let nextChar = message.charAt(index);

                // handle HTML tags
                if (nextChar === "<") {
                    const closeTag = message.indexOf(">", index);
                    wishEl.innerHTML += message.slice(index, closeTag + 1);
                    index = closeTag + 1;
                } else {
                    wishEl.innerHTML += nextChar;
                    index++;
                }

                // Very slow typing: ~2‚Äì3 minutes
                setTimeout(typeSlowly, 120); // adjust speed: 120ms per char ~ ~2.5 min
            }
        }

        function lightCandle() {
            document.querySelector(".cake").innerHTML = "üéÇüïØÔ∏è";

        function startCountdown() {
            let count = 3;
            const countdownEl = document.getElementById("countdown");
            countdownEl.innerHTML = count;

            const timer = setInterval(() => {
                count--;
                if (count > 0) countdownEl.innerHTML = count;
                else {
                    clearInterval(timer);
                    countdownEl.innerHTML = "üéâ";
                    revealWish();
                }
            }, 1000);
        }

        function revealWish() {
            document.getElementById("cakeSection").style.display = "none";
            wishEl.style.display = "block";
            music.play();
            confetti({ particleCount: 400, spread: 150, origin: { y: 0.6 } });

            setInterval(() => {

                heart.className = "heart";

        function startContinuousFireworks() {
            setInterval(() => {
                createFirework(Math.random() * window.innerWidth, Math.random() * window.innerHeight / 2);
            }, 800);
        }

        function createFirework(x, y) {
            const fireworkCount = 12;
            for (let i = 0; i < fireworkCount; i++) {
                const p = document.createElement("div");
                p.className = "firework";
                p.style.left = x + "px";
                p.style.top = y + "px";
                p.style.background = `hsl(${Math.random()*360}, 100%, 70%)`;
                p.style.width = "4px";
                p.style.height = "4px";
                document.body.appendChild(p);

                let angle = (i / fireworkCount) * 2 * Math.PI;
                let speed = Math.random() * 3 + 2;
                let step = 0;

                const anim = setInterval(() => {
                    step++;
                    const dx = Math.cos(angle) * speed * step;
                    const dy = Math.sin(angle) * speed * step;
                    p.style.left = x + dx + "px";
                    p.style.top = y + dy + "px";
                    p.style.opacity -= 0.02;
                    if (step > 30) {
                        clearInterval(anim);
                        p.remove();
                    }
                }, 30);
            }
        }
